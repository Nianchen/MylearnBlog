    # 执行上下文

    JS的独特性让他在执行代码的时候，会提前进行一部分的工作，例如变量提升，函数提升，这个提升其实也可以是看作，在开辟执行上下文 JS最外层的执行上下文就应该是全局执行上下文，也就是global stack，这个执行上下文会在所有代码执行结束后再进行释放。

    我们可以利用一个stack也就是栈取模拟这个函数上下文的情况
var ECStack = [globalContext]
//直接在里面放入全局上下文
function fun3() {
    console.log('fun3')
}
function fun2() {
    fun3();
}
function fun1() {
    fun2();
}
fun1();
//在这段代码的运行情况中
// 伪代码
// fun1()
ECStack.push(<fun1> functionContext);
// fun1中竟然调用了fun2，还要创建fun2的执行上下文
ECStack.push(<fun2> functionContext);
// 擦，fun2还调用了fun3！
ECStack.push(<fun3> functionContext);
// fun3执行完毕
ECStack.pop();
// fun2执行完毕
ECStack.pop();
// fun1执行完毕
ECStack.pop();
// javascript接着执行下面的代码，但是ECStack底层永远有个globalContext直到程序结束

    # 模拟实现call与apply

    call() 方法在使用 一个指定的 this值 和 若干个指定的参数值 的前提下调用 某个 函数 或 方法。

    eg
var foo = {
    value: 1
};

function bar() {
    console.log(this.value);
}
bar.call(foo); // 1
在这里的意思就是bar的this指向了foo
也可以理解为在foo中添加了一个bar的函数

    我们需要实现call 其实就是实现三个东西，1.改变this指向 2.call参数的解决 3.call能被所有函数调用

    首先这个call函数在所有函数上都能调用 所以我们直接定义在他的原型上
首先要明白一点，哪个函数被调用那么this就在哪
Function.prototype.call2 = function(context) {
    // 这个context就是传进来的foo
    // this就是调用这个函数的bar
    context.fn = this;
    context.fn();
    delete context.fn;
}
var foo = {
    value: 1
};
function bar() {
    console.log(this.value);
}
bar.call2(foo); // 1
//但是我们也知道call是可以传参数的
var foo = {
	value:1
}
function bar(name,age){
    console.log(name)
    console.log(age)
    console.log(this.value)
}
bar.call(foo,'Nianchen',20)

//20
//1
涉及到参数我们能很容易的想起来一个东西，argments不管你传入多少值，都可以在这个对象里取到
//这就是完整版的call
Function.prototype.Mycall = function(target,...args){
  target = target || window
  const symbolKey = Symbol()
  target[symbolKey] = this
  const res = target[symbolKey](...args) // args本身是rest参数，搭配的变量是一个数组，数组解构后就可以一个个传入函数中
  delete target[symbolKey] // 执行完借用的函数后，删除掉，留着过年吗？
  return res
}
//apply 跟call的差别在于 apply第二个参数传递的是一个数组
var person = {
    fullName: function() {
        return this.firstName + " " + this.lastName;
    }
}
var person1 = {
    firstName: "Bill",
    lastName: "Gates",
}
person.fullName.apply(person1);  // 将返回 "Bill Gates"

//实现
Function.prototype.apply = function (context, arr) {
    var context = Object(context) || window;
    context.fn = this;

    var result;
    if (!arr) {
        result = context.fn();
    }
    else {
        var args = [];
        for (var i = 0, len = arr.length; i < len; i++) {
            args.push('arr[' + i + ']');
        }
        result = eval('context.fn(' + args + ')')
    }
    delete context.fn
    return result;
}
